Vagrant.configure("2") do |config|
    config.vm.box = "spox/ubuntu-arm"

    config.vm.provider "vmware_fusion" do |v|
        v.memory = 1024
        v.cpus = 2
    end

    # Sync vagrant file in host machine 
    config.vm.synced_folder "./scripts", "/opt/scripts"
    config.vm.provision "shell", inline: <<-SHELL
        cd /opt/scripts
        touch /opt/scripts/provisioned.txt
    SHELL

    config.vm.network "private_network", ip: "192.168.56.17", bridge: nil
    config.vm.network "public_network", bridge: true
end
